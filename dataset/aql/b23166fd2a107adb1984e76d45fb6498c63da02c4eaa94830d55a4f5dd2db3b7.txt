SELECT * FROM events WHERE devicetype=12 AND (LOWER("Process Path") LIKE '%\sqlcmd.exe' OR LOWER("Process Name") LIKE '%\sqlcmd.exe' OR LOWER(Command) LIKE '%sqlcmd %' OR LOWER(Command) LIKE '%sqlcmd.exe%') AND LOWER(Command) LIKE '% -s localhost %' AND (LOWER(Command) LIKE '%sys.sysprocesses%' OR LOWER(Command) LIKE '%master.dbo.sysdatabases%' OR LOWER(Command) LIKE '%backup database%')
