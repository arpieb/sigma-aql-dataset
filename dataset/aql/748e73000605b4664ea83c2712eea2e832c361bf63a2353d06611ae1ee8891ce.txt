SELECT * FROM events WHERE devicetype=12 AND (LOWER("Process Path") LIKE '%\powershell.exe' OR LOWER("Process Path") LIKE '%\pwsh.exe' OR LOWER("Process Path") LIKE '%\wmic.exe' OR LOWER("Process Path") LIKE '%\vssadmin.exe' OR LOWER("Process Name") LIKE '%\powershell.exe' OR LOWER("Process Name") LIKE '%\pwsh.exe' OR LOWER("Process Name") LIKE '%\wmic.exe' OR LOWER("Process Name") LIKE '%\vssadmin.exe' OR Filename IN('PowerShell.EXE', 'pwsh.dll', 'wmic.exe', 'VSSADMIN.EXE')) AND LOWER(Command) LIKE '%shadow%' AND LOWER(Command) LIKE '%create%'
