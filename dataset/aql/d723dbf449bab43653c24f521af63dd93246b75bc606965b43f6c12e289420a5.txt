SELECT * FROM events WHERE devicetype=12 AND (LOWER("Process Path") LIKE '%:\windows\system32\bash.exe' OR LOWER("Process Path") LIKE '%:\windows\syswow64\bash.exe' OR LOWER("Process Name") LIKE '%:\windows\system32\bash.exe' OR LOWER("Process Name") LIKE '%:\windows\syswow64\bash.exe' OR Filename='Bash.exe') AND NOT((LOWER(Command) LIKE '%bash.exe -%' OR LOWER(Command) LIKE '%bash -%' OR Command IS NULL OR Command='' OR Command IN('bash.exe', 'bash')))
