SELECT * FROM events WHERE devicetype=12 AND (LOWER("Process Path") LIKE '%\reg.exe' OR LOWER("Process Name") LIKE '%\reg.exe' OR Filename='reg.exe') AND LOWER(Command) LIKE '% import %' AND (LOWER(Command) LIKE '%c:\users\%' OR LOWER(Command) LIKE '%%temp%%' OR LOWER(Command) LIKE '%%tmp%%' OR LOWER(Command) LIKE '%%appdata%%' OR LOWER(Command) LIKE '%\appdata\local\temp\%' OR LOWER(Command) LIKE '%c:\windows\temp\%' OR LOWER(Command) LIKE '%c:\programdata\%')
