SELECT * FROM events WHERE devicetype=102 AND (LOWER("Process Path") LIKE '%/dscl' OR LOWER("Process Name") LIKE '%/dscl') AND LOWER(Command) LIKE '%create%' AND LOWER(Command) LIKE '%uniqueid%' AND Command MATCHES '([0-9]|[1-9][0-9]|[1-4][0-9]{2})' OR (LOWER("Process Path") LIKE '%/dscl' OR LOWER("Process Name") LIKE '%/dscl') AND LOWER(Command) LIKE '%create%' AND LOWER(Command) LIKE '%ishidden%' AND (LOWER(Command) LIKE '%true%' OR LOWER(Command) LIKE '%yes%' OR Command LIKE '%1%')
