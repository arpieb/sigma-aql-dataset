SELECT * FROM events WHERE devicetype=12 AND (LOWER("Process Path") LIKE '%\csi.exe' OR LOWER("Process Name") LIKE '%\csi.exe') AND (LOWER("Parent Process Path") LIKE '%\powershell.exe' OR LOWER("Parent Process Path") LIKE '%\pwsh.exe' OR LOWER("Parent Process Path") LIKE '%\powershell_ise.exe') AND Filename='csi.exe'
