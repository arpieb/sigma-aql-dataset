SELECT * FROM events WHERE devicetype=12 AND (LOWER("Process Path") LIKE '%\schtasks.exe' OR LOWER("Process Name") LIKE '%\schtasks.exe' OR Filename='schtasks.exe') AND (LOWER(Command) LIKE '% onlogon %' OR LOWER(Command) LIKE '% onstart %' OR LOWER(Command) LIKE '% once %' OR LOWER(Command) LIKE '% onidle %') AND (LOWER(Command) LIKE '%nt aut%' OR LOWER(Command) LIKE '% system%' OR LOWER(Command) LIKE '%highest%')
