SELECT * FROM events WHERE devicetype=12 AND (Filename='HH.exe' OR LOWER("Process Path") LIKE '%\hh.exe' OR LOWER("Process Name") LIKE '%\hh.exe') AND (LOWER(Command) LIKE '%.application%' OR LOWER(Command) LIKE '%\appdata\local\temp\%' OR LOWER(Command) LIKE '%\content.outlook\%' OR LOWER(Command) LIKE '%\downloads\%' OR LOWER(Command) LIKE '%\users\public\%' OR LOWER(Command) LIKE '%\windows\temp\%')
