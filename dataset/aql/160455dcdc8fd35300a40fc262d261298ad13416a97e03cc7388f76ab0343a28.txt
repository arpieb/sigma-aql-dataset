SELECT *, Username, "Process Path", "Process Name" FROM events WHERE devicetype=12 AND LOWER("Parent Process Path") LIKE '%\winword.exe' AND (LOWER("Process Path") LIKE '%.tmp' OR LOWER("Process Name") LIKE '%.tmp') OR (LOWER("Process Path") LIKE '%\wmic.exe' OR LOWER("Process Name") LIKE '%\wmic.exe') AND LOWER("Parent Process Path") LIKE '%\temp\%' AND LOWER(Command) LIKE '%shadowcopy delete' OR LOWER(Command) LIKE '%shadowcopy delete' AND LOWER(Command) LIKE '%\..\..\system32%'
