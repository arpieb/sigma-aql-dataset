SELECT * FROM events WHERE devicetype=12 AND LOWER(Command) LIKE '%sc scheduled scan%' AND LOWER(Command) LIKE '%\microsoft\taskbar\autoit3.exe%' OR (LOWER("Process Path") LIKE '%\windows\temp\db\%' OR LOWER("Process Name") LIKE '%\windows\temp\db\%') AND (LOWER("Process Path") LIKE '%.exe' OR LOWER("Process Name") LIKE '%.exe') OR ("Process Path"='C:\Windows\system32\Service.exe' OR "Process Name"='C:\Windows\system32\Service.exe') AND (LOWER(Command) LIKE '%i%' OR LOWER(Command) LIKE '%u%') OR LOWER("Parent Process Path") LIKE '%\local\microsoft\taskbar\autoit3.exe' AND LOWER(Command) LIKE '%nslookup.exe%' AND LOWER(Command) LIKE '%-q=txt%'
