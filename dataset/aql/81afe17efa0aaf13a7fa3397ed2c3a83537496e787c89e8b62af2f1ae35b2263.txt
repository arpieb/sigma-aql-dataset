SELECT * FROM events WHERE devicetype=12 AND LOWER(Command) LIKE '%cmd.exe /c whoami' AND LOWER("Parent Process Path") LIKE 'c:\temp\%' OR (LOWER("Parent Process Path") LIKE '%\runonce.exe' OR LOWER("Parent Process Path") LIKE '%\dllhost.exe') AND LOWER(Command) LIKE '%cmd.exe /c echo%' AND LOWER(Command) LIKE '%> \\.\pipe%' OR LOWER("Parent Command") LIKE '%cmd.exe /c echo%' AND LOWER("Parent Command") LIKE '% > \\.\pipe%' AND LOWER(Command) LIKE '%conhost.exe 0xffffffff -forcev1' OR LOWER("Parent Command") LIKE '%/c whoami' AND LOWER(Command) LIKE '%conhost.exe 0xffffffff -forcev1'
