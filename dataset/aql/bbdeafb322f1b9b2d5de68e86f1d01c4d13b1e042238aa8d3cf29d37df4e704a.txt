SELECT * FROM events WHERE devicetype=11 AND LOWER("Parent Process Path") LIKE '%/java' AND LOWER("Parent Command") LIKE '%confluence%' AND (LOWER("Process Path") LIKE '%/bash' OR LOWER("Process Path") LIKE '%/curl' OR LOWER("Process Path") LIKE '%/echo' OR LOWER("Process Path") LIKE '%/wget' OR LOWER("Process Name") LIKE '%/bash' OR LOWER("Process Name") LIKE '%/curl' OR LOWER("Process Name") LIKE '%/echo' OR LOWER("Process Name") LIKE '%/wget') AND NOT(LOWER(Command) LIKE '%ulimit -u%')
