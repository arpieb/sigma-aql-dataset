SELECT * FROM events WHERE devicetype=12 AND (LOWER("Process Path") LIKE '%\msiexec.exe' OR LOWER("Process Name") LIKE '%\msiexec.exe' OR Filename='msiexec.exe') AND (LOWER(Command) LIKE '%/i%' OR LOWER(Command) LIKE '%-i%' OR LOWER(Command) LIKE '%/package%' OR LOWER(Command) LIKE '%-package%' OR LOWER(Command) LIKE '%/a%' OR LOWER(Command) LIKE '%-a%' OR LOWER(Command) LIKE '%/j%' OR LOWER(Command) LIKE '%-j%') AND (LOWER(Command) LIKE '%/q%' OR LOWER(Command) LIKE '%-q%') AND (LOWER(Command) LIKE '%http%' OR LOWER(Command) LIKE '%\\%')
