SELECT *, Command, "Parent Command" FROM events WHERE devicetype=12 AND LOWER("Process Path") LIKE '%\nircmd.exe' OR LOWER("Process Name") LIKE '%\nircmd.exe' OR Filename='NirCmd.exe' OR LOWER(Command) LIKE '% execmd %' OR LOWER(Command) LIKE '%.exe script %' OR LOWER(Command) LIKE '%.exe shexec %' OR LOWER(Command) LIKE '% runinteractive %' OR (LOWER(Command) LIKE '% exec %' OR LOWER(Command) LIKE '% exec2 %') AND (LOWER(Command) LIKE '% show %' OR LOWER(Command) LIKE '% hide %')
