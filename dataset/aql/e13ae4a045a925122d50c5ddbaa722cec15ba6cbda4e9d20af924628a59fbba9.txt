SELECT * FROM events WHERE devicetype=12 AND (LOWER("Parent Process Path") LIKE '%\wscript.exe' OR LOWER("Parent Process Path") LIKE '%\cscript.exe') AND (LOWER("Process Path") LIKE '%\cmd.exe' OR LOWER("Process Path") LIKE '%\powershell.exe' OR LOWER("Process Path") LIKE '%\pwsh.exe' OR LOWER("Process Name") LIKE '%\cmd.exe' OR LOWER("Process Name") LIKE '%\powershell.exe' OR LOWER("Process Name") LIKE '%\pwsh.exe') AND (LOWER(Command) LIKE '%mshta%' AND LOWER(Command) LIKE '%http%' OR LOWER(Command) LIKE '%rundll32%' OR LOWER(Command) LIKE '%regsvr32%' OR LOWER(Command) LIKE '%msiexec%')
