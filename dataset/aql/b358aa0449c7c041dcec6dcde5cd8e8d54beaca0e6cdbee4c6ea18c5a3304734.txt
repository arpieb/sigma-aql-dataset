SELECT * FROM events WHERE devicetype=12 AND (LOWER("Process Path") LIKE '%\cmd.exe' OR LOWER("Process Name") LIKE '%\cmd.exe' OR Filename='Cmd.Exe') AND (LOWER(Command) LIKE '%del %' OR LOWER(Command) LIKE '%erase %') AND (LOWER(Command) LIKE '% /f%' OR LOWER(Command) LIKE '% /s%' OR LOWER(Command) LIKE '% /q%')
