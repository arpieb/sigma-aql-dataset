SELECT * FROM events WHERE devicetype=12 AND (LOWER("Process Path") LIKE '%\powershell.exe' OR LOWER("Process Path") LIKE '%\pwsh.exe' OR LOWER("Process Path") LIKE '%\powershell_ise.exe' OR LOWER("Process Name") LIKE '%\powershell.exe' OR LOWER("Process Name") LIKE '%\pwsh.exe' OR LOWER("Process Name") LIKE '%\powershell_ise.exe' OR Filename IN('PowerShell.EXE', 'pwsh.dll')) AND LOWER(Command) LIKE '%set-netfirewallprofile %' AND LOWER(Command) LIKE '% -enabled %' AND LOWER(Command) LIKE '% false%' AND (LOWER(Command) LIKE '% -all %' OR LOWER(Command) LIKE '%public%' OR LOWER(Command) LIKE '%domain%' OR LOWER(Command) LIKE '%private%')
