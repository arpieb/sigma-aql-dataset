SELECT * FROM events WHERE devicetype=12 AND LOWER(Filename) LIKE '%\procmon%' AND LOWER(Filename) LIKE '%.sys' AND NOT((LOWER("Process Path") LIKE '%\procmon.exe' OR LOWER("Process Path") LIKE '%\procmon64.exe' OR LOWER("Process Name") LIKE '%\procmon.exe' OR LOWER("Process Name") LIKE '%\procmon64.exe'))
