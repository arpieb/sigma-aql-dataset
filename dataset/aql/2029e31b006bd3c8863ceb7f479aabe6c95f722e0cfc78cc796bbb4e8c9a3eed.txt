SELECT * FROM events WHERE devicetype=12 AND (LOWER("Process Path") LIKE '%\ping.exe' OR LOWER("Process Path") LIKE '%\arp.exe' OR LOWER("Process Name") LIKE '%\ping.exe' OR LOWER("Process Name") LIKE '%\arp.exe') AND (LOWER(Command) LIKE '% 0x%' OR LOWER(Command) LIKE '%//0x%' OR LOWER(Command) LIKE '%.0x%' OR LOWER(Command) LIKE '%.00x%' OR LOWER(Command) LIKE '%http://%%' AND LOWER(Command) LIKE '%%2e%' OR Command MATCHES 'https?://[0-9]{1,3}\.[0-9]{1,3}\.0[0-9]{3,4}' OR Command MATCHES 'https?://[0-9]{1,3}\.0[0-9]{3,7}' OR Command MATCHES 'https?://0[0-9]{3,11}' OR Command MATCHES 'https?://(0[0-9]{1,11}\.){3}0[0-9]{1,11}' OR Command MATCHES 'https?://0[0-9]{1,11}' OR Command MATCHES ' [0-7]{7,13}') AND NOT(Command MATCHES 'https?://((25[0-5]|(2[0-4]|1\d|[1-9])?\d)(\.|\b)){4}')
