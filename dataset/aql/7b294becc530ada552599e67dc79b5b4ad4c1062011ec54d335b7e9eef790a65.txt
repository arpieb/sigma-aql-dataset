SELECT * FROM events WHERE devicetype=12 AND (LOWER("Process Path") LIKE '%\certoc.exe' OR LOWER("Process Name") LIKE '%\certoc.exe' OR Filename='CertOC.exe') AND (LOWER(Command) LIKE '% -loaddll %' OR LOWER(Command) LIKE '% /loaddll %') AND (LOWER(Command) LIKE '%\appdata\local\temp\%' OR LOWER(Command) LIKE '%\desktop\%' OR LOWER(Command) LIKE '%\downloads\%' OR LOWER(Command) LIKE '%\users\public\%' OR LOWER(Command) LIKE '%c:\windows\tasks\%' OR LOWER(Command) LIKE '%c:\windows\temp\%')
