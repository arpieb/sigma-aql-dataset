SELECT * FROM events WHERE devicetype=12 AND (LOWER("Process Path") LIKE '%\rundll32.exe' OR LOWER("Process Name") LIKE '%\rundll32.exe' OR Filename='RUNDLL32.EXE') AND LOWER(Command) LIKE '%shell32.dll%' AND LOWER(Command) LIKE '%control_rundll%' AND (LOWER(Command) LIKE '%%appdata%%' OR LOWER(Command) LIKE '%%localappdata%%' OR LOWER(Command) LIKE '%%temp%%' OR LOWER(Command) LIKE '%%tmp%%' OR LOWER(Command) LIKE '%\appdata\%' OR LOWER(Command) LIKE '%\temp\%' OR LOWER(Command) LIKE '%\users\public\%')
