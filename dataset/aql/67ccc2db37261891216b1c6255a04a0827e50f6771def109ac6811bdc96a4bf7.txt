SELECT * FROM events WHERE devicetype=12 AND (LOWER("Parent Process Path") LIKE '%\powershell.exe' OR LOWER("Parent Process Path") LIKE '%\pwsh.exe') AND (LOWER("Process Path") LIKE '%\schtasks.exe' OR LOWER("Process Name") LIKE '%\schtasks.exe') AND LOWER(Command) LIKE '%/create%' AND LOWER(Command) LIKE '%powershell.exe -noni%' AND LOWER(Command) LIKE '%/tn updater /tr%' AND (LOWER(Command) LIKE '%/sc onlogon%' OR LOWER(Command) LIKE '%/sc daily /st%' OR LOWER(Command) LIKE '%/sc onidle%' OR LOWER(Command) LIKE '%/sc hourly%')
