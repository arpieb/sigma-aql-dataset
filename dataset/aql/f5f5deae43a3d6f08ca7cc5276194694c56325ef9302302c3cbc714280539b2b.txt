SELECT * FROM events WHERE devicetype=12 AND (LOWER("Process Path") LIKE '%\cmd.exe%' OR LOWER("Process Name") LIKE '%\cmd.exe%' OR Filename='Cmd.EXE') AND LOWER(Command) LIKE '% net use http%' AND LOWER(Command) LIKE '%& start /b %' AND LOWER(Command) LIKE '%\davwwwroot\%' AND (LOWER(Command) LIKE '%.exe %' OR LOWER(Command) LIKE '%.dll %' OR LOWER(Command) LIKE '%.bat %' OR LOWER(Command) LIKE '%.vbs %' OR LOWER(Command) LIKE '%.ps1 %')
