SELECT * FROM events WHERE devicetype=12 AND (LOWER("Process Path") LIKE '%\dismhost.exe' OR LOWER("Process Name") LIKE '%\dismhost.exe') AND LOWER("Parent Command") LIKE '%/online%' AND LOWER("Parent Command") LIKE '%/disable-feature%' OR (LOWER("Process Path") LIKE '%\dism.exe' OR LOWER("Process Name") LIKE '%\dism.exe') AND LOWER(Command) LIKE '%/online%' AND LOWER(Command) LIKE '%/disable-feature%'
