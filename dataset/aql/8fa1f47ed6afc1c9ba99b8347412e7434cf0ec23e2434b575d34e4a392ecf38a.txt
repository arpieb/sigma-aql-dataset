SELECT * FROM events WHERE devicetype=12 AND (LOWER("Process Path") LIKE '%\mshta.exe' OR LOWER("Process Name") LIKE '%\mshta.exe' OR Filename='MSHTA.EXE') AND (LOWER("Parent Process Path") LIKE '%\cmd.exe' OR LOWER("Parent Process Path") LIKE '%\cscript.exe' OR LOWER("Parent Process Path") LIKE '%\powershell.exe' OR LOWER("Parent Process Path") LIKE '%\pwsh.exe' OR LOWER("Parent Process Path") LIKE '%\regsvr32.exe' OR LOWER("Parent Process Path") LIKE '%\rundll32.exe' OR LOWER("Parent Process Path") LIKE '%\wscript.exe') AND (LOWER(Command) LIKE '%\appdata\local\%' OR LOWER(Command) LIKE '%c:\programdata\%' OR LOWER(Command) LIKE '%c:\users\public\%' OR LOWER(Command) LIKE '%c:\windows\temp\%') OR (LOWER("Process Path") LIKE '%\mshta.exe' OR LOWER("Process Name") LIKE '%\mshta.exe' OR Filename='MSHTA.EXE') AND NOT((LOWER("Process Path") LIKE 'c:\windows\system32\%' OR LOWER("Process Path") LIKE 'c:\windows\syswow64\%' OR LOWER("Process Name") LIKE 'c:\windows\system32\%' OR LOWER("Process Name") LIKE 'c:\windows\syswow64\%' OR LOWER(Command) LIKE '%.htm%' OR LOWER(Command) LIKE '%.hta%' OR LOWER(Command) LIKE '%mshta.exe' OR LOWER(Command) LIKE '%mshta'))
