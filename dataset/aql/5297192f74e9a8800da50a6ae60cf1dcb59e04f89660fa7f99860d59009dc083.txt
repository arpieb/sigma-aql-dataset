SELECT * FROM events WHERE devicetype=12 AND (Filename IN('PowerShell.EXE', 'pwsh.dll') OR LOWER("Process Path") LIKE '%\powershell.exe' OR LOWER("Process Path") LIKE '%\pwsh.exe' OR LOWER("Process Name") LIKE '%\powershell.exe' OR LOWER("Process Name") LIKE '%\pwsh.exe') AND LOWER(Command) LIKE '%set-acl %' AND LOWER(Command) LIKE '%-aclobject %' AND (LOWER(Command) LIKE '%-path "c:\windows%' OR LOWER(Command) LIKE '%-path ''c:\windows%' OR LOWER(Command) LIKE '%-path %windir%%' OR LOWER(Command) LIKE '%-path $env:windir%') AND (LOWER(Command) LIKE '%fullcontrol%' OR LOWER(Command) LIKE '%allow%')
