SELECT * FROM events WHERE devicetype=12 AND (LOWER("Process Path") LIKE '%\netsh.exe' OR LOWER("Process Name") LIKE '%\netsh.exe' OR Filename='netsh.exe') AND LOWER(Command) LIKE '%advfirewall%' AND LOWER(Command) LIKE '%firewall%' AND LOWER(Command) LIKE '%set%' AND LOWER(Command) LIKE '%rule%' AND LOWER(Command) LIKE '%group=%' AND LOWER(Command) LIKE '%new%' AND LOWER(Command) LIKE '%enable=yes%'
