SELECT * FROM events WHERE devicetype=12 AND (LOWER("Process Path") LIKE '%\mshta.exe' OR LOWER("Process Name") LIKE '%\mshta.exe') AND (LOWER(Command) LIKE '%vbscript%' OR LOWER(Command) LIKE '%.jpg%' OR LOWER(Command) LIKE '%.png%' OR LOWER(Command) LIKE '%.lnk%' OR LOWER(Command) LIKE '%.xls%' OR LOWER(Command) LIKE '%.doc%' OR LOWER(Command) LIKE '%.zip%' OR LOWER(Command) LIKE '%.dll%')
