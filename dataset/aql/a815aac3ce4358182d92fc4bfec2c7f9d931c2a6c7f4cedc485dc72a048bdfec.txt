SELECT * FROM events WHERE devicetype=12 AND (LOWER("Parent Process Path") LIKE '%\cmd.exe' OR LOWER("Parent Process Path") LIKE '%\cscript.exe' OR LOWER("Parent Process Path") LIKE '%\mshta.exe' OR LOWER("Parent Process Path") LIKE '%\powershell.exe' OR LOWER("Parent Process Path") LIKE '%\pwsh.exe' OR LOWER("Parent Process Path") LIKE '%\regsvr32.exe' OR LOWER("Parent Process Path") LIKE '%\rundll32.exe' OR LOWER("Parent Process Path") LIKE '%\schtasks.exe' OR LOWER("Parent Process Path") LIKE '%\wmic.exe' OR LOWER("Parent Process Path") LIKE '%\wscript.exe' OR LOWER("Parent Process Path") LIKE '%\wsl.exe') AND (LOWER("Process Path") LIKE '%\msdt.exe' OR LOWER("Process Name") LIKE '%\msdt.exe' OR Filename='msdt.exe')
