SELECT *, Filename, "Process Path", "Process Name" FROM events WHERE devicetype=12 AND NOT((LOWER("Process Path") LIKE 'c:\windows\system32\%' OR LOWER("Process Path") LIKE 'c:\windows\syswow64\%' OR LOWER("Process Name") LIKE 'c:\windows\system32\%' OR LOWER("Process Name") LIKE 'c:\windows\syswow64\%')) AND LOWER("Parent Command") LIKE '%cmd.exe /c%' AND LOWER("Parent Command") LIKE '%roamdiag.cmd%' AND LOWER("Parent Command") LIKE '%-outputpath%'
