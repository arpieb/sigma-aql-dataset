SELECT * FROM events WHERE devicetype=12 AND (LOWER("Process Path") LIKE '%\whoami.exe' OR LOWER("Process Name") LIKE '%\whoami.exe' OR Filename='whoami.exe') AND (LOWER(Command) LIKE '% -all%' OR LOWER(Command) LIKE '% /all%' OR LOWER(Command) LIKE '% /fo csv%' OR LOWER(Command) LIKE '% -fo csv%') OR LOWER(Command) LIKE '%whoami%>%'
