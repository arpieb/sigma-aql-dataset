SELECT * FROM events WHERE devicetype=12 AND (LOWER(Command) LIKE '%findstr%' OR LOWER("Process Path") LIKE '%findstr.exe' OR LOWER("Process Name") LIKE '%findstr.exe' OR Filename='FINDSTR.EXE') AND (LOWER(Command) LIKE '% /v %' OR LOWER(Command) LIKE '% -v %') AND (LOWER(Command) LIKE '% /l %' OR LOWER(Command) LIKE '% -l %') AND LOWER(Command) LIKE '%\\%'
