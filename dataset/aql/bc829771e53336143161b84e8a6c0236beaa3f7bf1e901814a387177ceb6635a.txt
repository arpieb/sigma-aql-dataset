SELECT * FROM events WHERE devicetype=12 AND (Filename='procdump' OR (LOWER(Command) LIKE '% -ma %' OR LOWER(Command) LIKE '% /ma %') AND (LOWER(Command) LIKE '% -accepteula %' OR LOWER(Command) LIKE '% /accepteula %')) AND NOT((LOWER("Process Path") LIKE '%\procdump.exe' OR LOWER("Process Path") LIKE '%\procdump64.exe' OR LOWER("Process Name") LIKE '%\procdump.exe' OR LOWER("Process Name") LIKE '%\procdump64.exe'))
