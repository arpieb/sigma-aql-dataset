SELECT * FROM events WHERE devicetype=12 AND (LOWER("Process Path") LIKE '%\powershell.exe' OR LOWER("Process Path") LIKE '%\pwsh.exe' OR LOWER("Process Name") LIKE '%\powershell.exe' OR LOWER("Process Name") LIKE '%\pwsh.exe' OR Filename IN('PowerShell.EXE', 'pwsh.dll')) AND (LOWER(Command) LIKE '%-disablebehaviormonitoring $true%' OR LOWER(Command) LIKE '%-disableruntimemonitoring $true%') OR (LOWER("Process Path") LIKE '%\sc.exe' OR LOWER("Process Name") LIKE '%\sc.exe' OR Filename='sc.exe') AND (LOWER(Command) LIKE '%stop%' AND LOWER(Command) LIKE '%windefend%' OR LOWER(Command) LIKE '%delete%' AND LOWER(Command) LIKE '%windefend%' OR LOWER(Command) LIKE '%config%' AND LOWER(Command) LIKE '%windefend%' AND LOWER(Command) LIKE '%start=disabled%')
