SELECT *, Username, Command FROM events WHERE devicetype=12 AND LOWER("Parent Process Path") LIKE 'c:\users\public\%' AND (LOWER(Command) LIKE '%powershell%' OR LOWER(Command) LIKE '%cmd.exe /c %' OR LOWER(Command) LIKE '%cmd.exe /r %' OR LOWER(Command) LIKE '%cmd.exe /k %' OR LOWER(Command) LIKE '%cmd /c %' OR LOWER(Command) LIKE '%cmd /r %' OR LOWER(Command) LIKE '%cmd /k %' OR LOWER(Command) LIKE '%wscript.exe%' OR LOWER(Command) LIKE '%cscript.exe%' OR LOWER(Command) LIKE '%bitsadmin%' OR LOWER(Command) LIKE '%certutil%' OR LOWER(Command) LIKE '%mshta.exe%')
