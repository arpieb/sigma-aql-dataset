SELECT *, Username, Command FROM events WHERE devicetype=12 AND (LOWER("Process Path") LIKE '%\cmstp.exe' OR LOWER("Process Name") LIKE '%\cmstp.exe' OR Filename='CMSTP.EXE') AND (LOWER(Command) LIKE '%/s%' OR LOWER(Command) LIKE '%-s%' OR LOWER(Command) LIKE '%/au%' OR LOWER(Command) LIKE '%-au%' OR LOWER(Command) LIKE '%/ni%' OR LOWER(Command) LIKE '%-ni%')
