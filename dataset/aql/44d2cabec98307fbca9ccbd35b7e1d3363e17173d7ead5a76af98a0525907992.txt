SELECT * FROM events WHERE devicetype=12 AND ((LOWER("Process Path") LIKE '%\cmd.exe' OR LOWER("Process Name") LIKE '%\cmd.exe') AND LOWER(Command) LIKE '%copy %' OR (LOWER("Process Path") LIKE '%\powershell.exe' OR LOWER("Process Path") LIKE '%\pwsh.exe' OR LOWER("Process Name") LIKE '%\powershell.exe' OR LOWER("Process Name") LIKE '%\pwsh.exe') AND (LOWER(Command) LIKE '%copy-item%' OR LOWER(Command) LIKE '% copy %' OR LOWER(Command) LIKE '%cpi %' OR LOWER(Command) LIKE '% cp %') OR LOWER("Process Path") LIKE '%\robocopy.exe' OR LOWER("Process Path") LIKE '%\xcopy.exe' OR LOWER("Process Name") LIKE '%\robocopy.exe' OR LOWER("Process Name") LIKE '%\xcopy.exe' OR Filename IN('robocopy.exe', 'XCOPY.EXE')) AND (LOWER(Command) LIKE '%\system32%' OR LOWER(Command) LIKE '%\syswow64%' OR LOWER(Command) LIKE '%\winsxs%')
