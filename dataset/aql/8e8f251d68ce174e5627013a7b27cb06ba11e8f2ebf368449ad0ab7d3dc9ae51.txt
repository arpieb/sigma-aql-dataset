SELECT *, Command, "Parent Command" FROM events WHERE devicetype=12 AND (LOWER("Process Path") LIKE '%\wwwroot\%' OR LOWER("Process Path") LIKE '%\wmpub\%' OR LOWER("Process Path") LIKE '%\htdocs\%' OR LOWER("Process Name") LIKE '%\wwwroot\%' OR LOWER("Process Name") LIKE '%\wmpub\%' OR LOWER("Process Name") LIKE '%\htdocs\%') AND NOT(((LOWER("Process Path") LIKE '%bin\%' OR LOWER("Process Path") LIKE '%\tools\%' OR LOWER("Process Path") LIKE '%\smscomponent\%' OR LOWER("Process Name") LIKE '%bin\%' OR LOWER("Process Name") LIKE '%\tools\%' OR LOWER("Process Name") LIKE '%\smscomponent\%') AND LOWER("Parent Process Path") LIKE '%\services.exe'))
