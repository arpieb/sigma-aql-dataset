SELECT *, Command, "Parent Command" FROM events WHERE devicetype=12 AND (LOWER("Parent Process Path") LIKE '%\winword.exe' OR LOWER("Parent Process Path") LIKE '%\excel.exe' OR LOWER("Parent Process Path") LIKE '%\powerpnt.exe' OR LOWER("Parent Process Path") LIKE '%\mspub.exe' OR LOWER("Parent Process Path") LIKE '%\visio.exe' OR LOWER("Parent Process Path") LIKE '%\msaccess.exe' OR LOWER("Parent Process Path") LIKE '%\eqnedt32.exe') AND (LOWER("Process Path") LIKE 'c:\users\%' OR LOWER("Process Name") LIKE 'c:\users\%') AND (LOWER("Process Path") LIKE '%.exe' OR LOWER("Process Name") LIKE '%.exe') AND NOT((LOWER("Process Path") LIKE '%\teams.exe' OR LOWER("Process Name") LIKE '%\teams.exe'))
