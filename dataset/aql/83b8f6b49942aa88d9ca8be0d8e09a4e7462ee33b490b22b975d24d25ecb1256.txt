SELECT * FROM events WHERE devicetype=12 AND "Parent Process Path"='C:\Windows\System32\OpenSSH\sshd.exe' OR (LOWER("Process Path") LIKE '%\ssh.exe' OR LOWER("Process Name") LIKE '%\ssh.exe') AND (LOWER(Command) LIKE '%proxycommand=%' OR LOWER(Command) LIKE '%permitlocalcommand%' AND LOWER(Command) LIKE '%localcommand%')
