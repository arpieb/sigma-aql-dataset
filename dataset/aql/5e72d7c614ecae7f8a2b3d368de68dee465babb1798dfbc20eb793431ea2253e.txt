SELECT * FROM events WHERE devicetype=12 AND (Filename='OpenConsole.exe' OR LOWER("Process Path") LIKE '%\openconsole.exe' OR LOWER("Process Name") LIKE '%\openconsole.exe') AND NOT((LOWER("Process Path") LIKE 'c:\program files\windowsapps\microsoft.windowsterminal%' OR LOWER("Process Name") LIKE 'c:\program files\windowsapps\microsoft.windowsterminal%'))
