SELECT * FROM events WHERE devicetype=12 AND (LOWER("Process Path") LIKE '%\wmic.exe' OR LOWER("Process Name") LIKE '%\wmic.exe') AND (LOWER(Command) LIKE '%/format%' OR LOWER(Command) LIKE '%-format%') AND NOT((LOWER(Command) LIKE '%format:list%' OR LOWER(Command) LIKE '%format:htable%' OR LOWER(Command) LIKE '%format:hform%' OR LOWER(Command) LIKE '%format:table%' OR LOWER(Command) LIKE '%format:mof%' OR LOWER(Command) LIKE '%format:value%' OR LOWER(Command) LIKE '%format:rawxml%' OR LOWER(Command) LIKE '%format:xml%' OR LOWER(Command) LIKE '%format:csv%'))
