SELECT * FROM events WHERE devicetype=12 AND (Filename IN('wscript.exe', 'cscript.exe') OR LOWER("Process Path") LIKE '%\wscript.exe' OR LOWER("Process Path") LIKE '%\cscript.exe' OR LOWER("Process Name") LIKE '%\wscript.exe' OR LOWER("Process Name") LIKE '%\cscript.exe') AND (LOWER(Command) LIKE '%.js%' OR LOWER(Command) LIKE '%.jse%' OR LOWER(Command) LIKE '%.vba%' OR LOWER(Command) LIKE '%.vbe%' OR LOWER(Command) LIKE '%.vbs%' OR LOWER(Command) LIKE '%.wsf%')
