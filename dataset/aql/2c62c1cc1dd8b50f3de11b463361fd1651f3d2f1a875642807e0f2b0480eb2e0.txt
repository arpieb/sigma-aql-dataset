SELECT *, "Parent Process Path", Command FROM events WHERE devicetype=12 AND (LOWER("Process Path") LIKE '%\regedit.exe' OR LOWER("Process Name") LIKE '%\regedit.exe' OR Filename='REGEDIT.EXE') AND (LOWER(Command) LIKE '% /i %' OR LOWER(Command) LIKE '%.reg%') AND Command MATCHES ':[^ \\]' AND NOT((LOWER(Command) LIKE '% /e %' OR LOWER(Command) LIKE '% /a %' OR LOWER(Command) LIKE '% /c %' OR LOWER(Command) LIKE '% -e %' OR LOWER(Command) LIKE '% -a %' OR LOWER(Command) LIKE '% -c %'))
