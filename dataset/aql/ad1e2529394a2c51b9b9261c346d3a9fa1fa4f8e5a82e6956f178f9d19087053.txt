SELECT * FROM events WHERE devicetype=12 AND (LOWER(Command) LIKE '%findstr%' OR LOWER("Process Path") LIKE '%findstr.exe' OR LOWER("Process Name") LIKE '%findstr.exe' OR Filename='FINDSTR.EXE') AND (LOWER(Command) LIKE '% /s %' OR LOWER(Command) LIKE '% -s %') AND (LOWER(Command) LIKE '% /i %' OR LOWER(Command) LIKE '% -i %')
