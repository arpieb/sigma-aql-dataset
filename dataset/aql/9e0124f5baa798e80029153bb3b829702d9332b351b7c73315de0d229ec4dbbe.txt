SELECT * FROM events WHERE devicetype=12 AND (LOWER("Process Path") LIKE '%\svchost.exe' OR LOWER("Process Name") LIKE '%\svchost.exe') AND LOWER(Command) LIKE '%c:\users\%' AND LOWER(Command) LIKE '%\desktop\%' AND NOT(LOWER("Parent Process Path") LIKE 'c:\windows\system32\%') OR LOWER("Parent Process Path") LIKE '%\excel.exe' AND (LOWER("Process Path") LIKE '%\regsvr32.exe' OR LOWER("Process Name") LIKE '%\regsvr32.exe') AND (LOWER(Command) LIKE '% -s %' OR LOWER(Command) LIKE '%\appdata\local\temp\%') AND NOT(LOWER(Command) LIKE '%.dll%') OR LOWER("Parent Process Path") LIKE '%\svchost.exe' AND ((LOWER("Process Path") LIKE '%\whoami.exe' OR LOWER("Process Name") LIKE '%\whoami.exe') AND LOWER(Command) LIKE '% /all%' OR (LOWER("Process Path") LIKE '%\net.exe' OR LOWER("Process Path") LIKE '%\net1.exe' OR LOWER("Process Name") LIKE '%\net.exe' OR LOWER("Process Name") LIKE '%\net1.exe') AND LOWER(Command) LIKE '% view%')
