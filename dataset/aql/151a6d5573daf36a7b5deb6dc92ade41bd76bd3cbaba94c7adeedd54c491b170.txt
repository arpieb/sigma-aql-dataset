SELECT * FROM events WHERE devicetype=12 AND (LOWER("Process Path") LIKE '%\powershell.exe' OR LOWER("Process Path") LIKE '%\pwsh.exe' OR LOWER("Process Name") LIKE '%\powershell.exe' OR LOWER("Process Name") LIKE '%\pwsh.exe') AND LOWER(Filename) LIKE '%.ps1' AND NOT((LOWER(Filename) LIKE '%__psscriptpolicytest_%' OR LOWER(Filename) LIKE 'c:\users\%' AND LOWER(Filename) LIKE '%\appdata\local\temp\%' OR LOWER(Filename) LIKE 'c:\windows\temp\%'))
