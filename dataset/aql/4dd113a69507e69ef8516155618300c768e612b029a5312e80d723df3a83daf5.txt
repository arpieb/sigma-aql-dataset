SELECT * FROM events WHERE devicetype=12 AND (LOWER("Process Path") LIKE '%\control.exe' OR LOWER("Process Name") LIKE '%\control.exe') AND (LOWER("Parent Process Path") LIKE '%\winword.exe' OR LOWER("Parent Process Path") LIKE '%\powerpnt.exe' OR LOWER("Parent Process Path") LIKE '%\excel.exe') AND NOT((LOWER(Command) LIKE '%\control.exe input.dll' OR LOWER(Command) LIKE '%\control.exe" input.dll'))
