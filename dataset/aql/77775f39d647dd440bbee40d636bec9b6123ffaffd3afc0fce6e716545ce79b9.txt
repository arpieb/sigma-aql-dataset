SELECT * FROM events WHERE devicetype=12 AND (LOWER("Process Path") LIKE '%\rundll32.exe' OR LOWER("Process Name") LIKE '%\rundll32.exe') AND (LOWER(Command) LIKE '%%localappdata%%' OR LOWER(Command) LIKE '%\appdata\local\%') AND (LOWER(Command) LIKE '%.dat",%' OR LOWER(Command) LIKE '%.dll #1' OR LOWER(Command) LIKE '%.dll" #1' OR LOWER(Command) LIKE '%.dll",#1') AND NOT(LOWER(Command) LIKE '%\appdata\local\temp\%')
