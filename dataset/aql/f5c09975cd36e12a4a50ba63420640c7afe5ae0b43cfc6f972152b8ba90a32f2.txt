SELECT * FROM events WHERE devicetype=12 AND (Filename='wmic.exe' OR LOWER("Process Path") LIKE '%\wmic.exe' OR LOWER("Process Name") LIKE '%\wmic.exe') AND LOWER(Command) LIKE '% service %' AND LOWER(Command) LIKE '% call %' AND (LOWER(Command) LIKE '%stopservice%' OR LOWER(Command) LIKE '%startservice%')
