SELECT *, Command, "Parent Command" FROM events WHERE devicetype=12 AND (LOWER("Parent Process Path") LIKE '%\wmiprvse.exe' OR LOWER("Parent Process Path") LIKE '%\mmc.exe' OR LOWER("Parent Process Path") LIKE '%\explorer.exe' OR LOWER("Parent Process Path") LIKE '%\services.exe') AND LOWER(Command) LIKE '%cmd.exe%' AND LOWER(Command) LIKE '%/q%' AND LOWER(Command) LIKE '%/c%' AND LOWER(Command) LIKE '%\\127.0.0.1\%' AND LOWER(Command) LIKE '%&1%' OR (LOWER("Parent Command") LIKE '%svchost.exe -k netsvcs%' OR LOWER("Parent Command") LIKE '%taskeng.exe%') AND LOWER(Command) LIKE '%cmd.exe%' AND LOWER(Command) LIKE '%/c%' AND LOWER(Command) LIKE '%windows\temp\%' AND LOWER(Command) LIKE '%&1%'
