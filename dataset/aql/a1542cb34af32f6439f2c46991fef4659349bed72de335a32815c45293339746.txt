SELECT * FROM events WHERE devicetype=11 AND (LOWER("Process Path") LIKE '%/file' OR LOWER("Process Name") LIKE '%/file') AND Command MATCHES '(.){200,}' OR (LOWER("Process Path") LIKE '%/ls' OR LOWER("Process Name") LIKE '%/ls') AND LOWER(Command) LIKE '%-r%' OR LOWER("Process Path") LIKE '%/find' OR LOWER("Process Name") LIKE '%/find' OR LOWER("Process Path") LIKE '%/tree' OR LOWER("Process Name") LIKE '%/tree'
