SELECT *, Command, "Process Path", "Process Name", "Parent Command" FROM events WHERE devicetype=12 AND LOWER("Parent Process Path") LIKE '%\mmc.exe' AND (LOWER("Process Path") LIKE '%\cmd.exe' OR LOWER("Process Path") LIKE '%\powershell.exe' OR LOWER("Process Path") LIKE '%\pwsh.exe' OR LOWER("Process Path") LIKE '%\wscript.exe' OR LOWER("Process Path") LIKE '%\cscript.exe' OR LOWER("Process Path") LIKE '%\sh.exe' OR LOWER("Process Path") LIKE '%\bash.exe' OR LOWER("Process Path") LIKE '%\reg.exe' OR LOWER("Process Path") LIKE '%\regsvr32.exe' OR LOWER("Process Name") LIKE '%\cmd.exe' OR LOWER("Process Name") LIKE '%\powershell.exe' OR LOWER("Process Name") LIKE '%\pwsh.exe' OR LOWER("Process Name") LIKE '%\wscript.exe' OR LOWER("Process Name") LIKE '%\cscript.exe' OR LOWER("Process Name") LIKE '%\sh.exe' OR LOWER("Process Name") LIKE '%\bash.exe' OR LOWER("Process Name") LIKE '%\reg.exe' OR LOWER("Process Name") LIKE '%\regsvr32.exe' OR LOWER("Process Path") LIKE '%\bitsadmin%' OR LOWER("Process Name") LIKE '%\bitsadmin%')
