SELECT * FROM events WHERE devicetype=12 AND (LOWER("Process Path") LIKE '%\adplus.exe' OR LOWER("Process Name") LIKE '%\adplus.exe' OR Filename='Adplus.exe') AND (LOWER(Command) LIKE '% -hang %' OR LOWER(Command) LIKE '% -pn %' OR LOWER(Command) LIKE '% -pmn %' OR LOWER(Command) LIKE '% -p %' OR LOWER(Command) LIKE '% -po %' OR LOWER(Command) LIKE '% -c %' OR LOWER(Command) LIKE '% -sc %')
