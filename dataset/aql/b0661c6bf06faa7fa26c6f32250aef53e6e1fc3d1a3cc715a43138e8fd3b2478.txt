SELECT * FROM events WHERE devicetype=12 AND (LOWER("Process Path") LIKE '%\netsh.exe' OR LOWER("Process Name") LIKE '%\netsh.exe' OR Filename='netsh.exe') AND LOWER(Command) LIKE '%netsh %' AND LOWER(Command) LIKE '%show %' AND LOWER(Command) LIKE '%firewall %' AND (LOWER(Command) LIKE '%config %' OR LOWER(Command) LIKE '%state %' OR LOWER(Command) LIKE '%rule %' OR LOWER(Command) LIKE '%name=all%')
