SELECT * FROM events WHERE devicetype=12 AND (Filename='diskshadow.exe' OR LOWER("Process Path") LIKE '%\diskshadow.exe' OR LOWER("Process Name") LIKE '%\diskshadow.exe') AND (LOWER(Command) LIKE '%/s %' OR LOWER(Command) LIKE '%-s %') AND (LOWER(Command) LIKE '%:\temp\%' OR LOWER(Command) LIKE '%:\windows\temp\%' OR LOWER(Command) LIKE '%\appdata\local\%' OR LOWER(Command) LIKE '%\appdata\roaming\%' OR LOWER(Command) LIKE '%\programdata\%' OR LOWER(Command) LIKE '%\users\public\%')
