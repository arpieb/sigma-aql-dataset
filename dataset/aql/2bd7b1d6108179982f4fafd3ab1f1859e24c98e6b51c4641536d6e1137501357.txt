SELECT * FROM events WHERE devicetype=12 AND (LOWER("Process Path") LIKE '%\excel.exe' OR LOWER("Process Path") LIKE '%\powerpnt.exe' OR LOWER("Process Path") LIKE '%\winword.exe' OR LOWER("Process Name") LIKE '%\excel.exe' OR LOWER("Process Name") LIKE '%\powerpnt.exe' OR LOWER("Process Name") LIKE '%\winword.exe' OR Filename IN('Excel.exe', 'POWERPNT.EXE', 'WinWord.exe')) AND (LOWER(Command) LIKE '%http://%' OR LOWER(Command) LIKE '%https://%')
