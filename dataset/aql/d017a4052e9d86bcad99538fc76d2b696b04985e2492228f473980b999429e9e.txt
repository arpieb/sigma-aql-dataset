SELECT *, Username, Command FROM events WHERE devicetype=12 AND LOWER("Parent Process Path") LIKE '%\services.exe' AND (LOWER(Command) LIKE '%/c%' AND LOWER(Command) LIKE '%echo%' AND LOWER(Command) LIKE '%\pipe\%' AND (LOWER(Command) LIKE '%cmd%' OR LOWER(Command) LIKE '%%comspec%%') OR LOWER(Command) LIKE '%rundll32%' AND LOWER(Command) LIKE '%.dll,a%' AND LOWER(Command) LIKE '%/p:%') AND NOT(LOWER(Command) LIKE '%mpcmdrun%')
