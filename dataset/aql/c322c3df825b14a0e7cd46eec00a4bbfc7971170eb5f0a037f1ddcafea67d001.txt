SELECT * FROM events WHERE devicetype=12 AND (LOWER("Process Path") LIKE '%\powershell.exe' OR LOWER("Process Name") LIKE '%\powershell.exe') AND (LOWER(Command) LIKE '% -version 2 %' OR LOWER(Command) LIKE '% -versio 2 %' OR LOWER(Command) LIKE '% -versi 2 %' OR LOWER(Command) LIKE '% -vers 2 %' OR LOWER(Command) LIKE '% -ver 2 %' OR LOWER(Command) LIKE '% -ve 2 %' OR LOWER(Command) LIKE '% -v 2 %')
