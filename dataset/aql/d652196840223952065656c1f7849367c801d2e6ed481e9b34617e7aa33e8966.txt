SELECT * FROM events WHERE devicetype=12 AND (LOWER("Process Path") LIKE '%\wsl.exe' OR LOWER("Process Name") LIKE '%\wsl.exe' OR Filename='wsl.exe') AND (LOWER(Command) LIKE '% -e %' OR LOWER(Command) LIKE '% --exec%' OR LOWER(Command) LIKE '% --system%' OR LOWER(Command) LIKE '% --shell-type %' OR LOWER(Command) LIKE '% /mnt/c%' OR LOWER(Command) LIKE '% --user root%' OR LOWER(Command) LIKE '% -u root%' OR LOWER(Command) LIKE '%--debug-shell%') AND NOT((LOWER("Parent Process Path") LIKE '%\cmd.exe' AND LOWER(Command) LIKE '% -d %' AND LOWER(Command) LIKE '% -e kill %'))
