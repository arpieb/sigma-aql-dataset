SELECT * FROM events WHERE devicetype=12 AND (LOWER(Filename) LIKE '7z%.exe' OR LOWER(Filename) LIKE '%rar.exe' OR LOWER(Filename) LIKE '%command%line%rar%') AND (LOWER(Command) LIKE '% -p%' OR LOWER(Command) LIKE '% -ta%' OR LOWER(Command) LIKE '% -tb%' OR LOWER(Command) LIKE '% -sdel%' OR LOWER(Command) LIKE '% -dw%' OR LOWER(Command) LIKE '% -hp%') AND NOT((LOWER("Parent Process Path") LIKE '%:\program files\%' OR LOWER("Parent Process Path") LIKE '%:\program files (x86)\%'))
