SELECT * FROM events WHERE devicetype=12 AND (LOWER("Process Path") LIKE '%\powershell.exe' OR LOWER("Process Path") LIKE '%\pwsh.exe' OR LOWER("Process Name") LIKE '%\powershell.exe' OR LOWER("Process Name") LIKE '%\pwsh.exe' OR Filename IN('PowerShell.EXE', 'pwsh.dll')) AND (LOWER(Command) LIKE '%curl %' OR LOWER(Command) LIKE '%invoke-webrequest%' OR LOWER(Command) LIKE '%iwr %' OR LOWER(Command) LIKE '%wget %') AND (LOWER(Command) LIKE '% -ur%' OR LOWER(Command) LIKE '% -o%') AND (LOWER(Command) LIKE '%\appdata\%' OR LOWER(Command) LIKE '%\desktop\%' OR LOWER(Command) LIKE '%\temp\%' OR LOWER(Command) LIKE '%\users\public\%' OR LOWER(Command) LIKE '%%appdata%%' OR LOWER(Command) LIKE '%%public%%' OR LOWER(Command) LIKE '%%temp%%' OR LOWER(Command) LIKE '%%tmp%%' OR LOWER(Command) LIKE '%c:\windows\%')
