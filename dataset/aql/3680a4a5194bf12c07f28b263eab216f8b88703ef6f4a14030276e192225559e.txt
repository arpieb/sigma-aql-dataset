SELECT * FROM events WHERE devicetype=12 AND LOWER("Parent Process Path") LIKE '%\wscript.exe' AND (LOWER("Process Path") LIKE '%\cmd.exe' OR LOWER("Process Name") LIKE '%\cmd.exe') AND LOWER(Command) LIKE '%>>%appdata%\microsoft\%' AND (LOWER(Command) LIKE '%.xml' OR LOWER(Command) LIKE '%.txt') AND (LOWER(Command) LIKE '%dir%' OR LOWER(Command) LIKE '%ipconfig /all%' OR LOWER(Command) LIKE '%systeminfo%' OR LOWER(Command) LIKE '%tasklist%')
