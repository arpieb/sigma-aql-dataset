SELECT * FROM events WHERE devicetype=12 AND LOWER("Parent Process Path") LIKE '%\cmd.exe' AND LOWER("Parent Command") LIKE '%.bat' AND ((LOWER("Process Path") LIKE '%\xcopy.exe' OR LOWER("Process Name") LIKE '%\xcopy.exe') AND LOWER(Command) LIKE '%powershell.exe%' AND LOWER(Command) LIKE '%.bat.exe%' OR (LOWER("Process Path") LIKE '%\xcopy.exe' OR LOWER("Process Name") LIKE '%\xcopy.exe') AND LOWER(Command) LIKE '%pwsh.exe%' AND LOWER(Command) LIKE '%.bat.exe%' OR (LOWER("Process Path") LIKE '%\attrib.exe' OR LOWER("Process Name") LIKE '%\attrib.exe') AND LOWER(Command) LIKE '%+s%' AND LOWER(Command) LIKE '%+h%' AND LOWER(Command) LIKE '%.bat.exe%')
