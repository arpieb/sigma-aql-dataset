SELECT * FROM events WHERE devicetype=12 AND ((LOWER("Process Path") LIKE '%\winword.exe' OR LOWER("Process Name") LIKE '%\winword.exe') AND LOWER(Filename) LIKE '%\windows\inetcache%' AND LOWER(Filename) LIKE '%.cab' OR (LOWER("Process Path") LIKE '%\winword.exe' OR LOWER("Process Name") LIKE '%\winword.exe') AND LOWER(Filename) LIKE '%\appdata\local\temp\%' AND LOWER(Filename) LIKE '%.inf%') AND NOT((LOWER(Filename) LIKE 'c:\users\%' AND LOWER(Filename) LIKE '%appdata\local\temp%' AND LOWER(Filename) LIKE '%\content.inf'))
