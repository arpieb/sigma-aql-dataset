SELECT *, Command, "Parent Command" FROM events WHERE devicetype=12 AND (LOWER("Process Path") LIKE '%\powershell.exe' OR LOWER("Process Path") LIKE '%\pwsh.exe' OR LOWER("Process Path") LIKE '%\wmic.exe' OR LOWER("Process Path") LIKE '%\vssadmin.exe' OR LOWER("Process Path") LIKE '%\diskshadow.exe' OR LOWER("Process Name") LIKE '%\powershell.exe' OR LOWER("Process Name") LIKE '%\pwsh.exe' OR LOWER("Process Name") LIKE '%\wmic.exe' OR LOWER("Process Name") LIKE '%\vssadmin.exe' OR LOWER("Process Name") LIKE '%\diskshadow.exe' OR Filename IN('PowerShell.EXE', 'pwsh.dll', 'wmic.exe', 'VSSADMIN.EXE', 'diskshadow.exe')) AND LOWER(Command) LIKE '%shadow%' AND LOWER(Command) LIKE '%delete%' OR (LOWER("Process Path") LIKE '%\wbadmin.exe' OR LOWER("Process Name") LIKE '%\wbadmin.exe' OR Filename='WBADMIN.EXE') AND LOWER(Command) LIKE '%delete%' AND LOWER(Command) LIKE '%catalog%' AND LOWER(Command) LIKE '%quiet%' OR (LOWER("Process Path") LIKE '%\vssadmin.exe' OR LOWER("Process Name") LIKE '%\vssadmin.exe' OR Filename='VSSADMIN.EXE') AND LOWER(Command) LIKE '%resize%' AND LOWER(Command) LIKE '%shadowstorage%' AND (LOWER(Command) LIKE '%unbounded%' OR LOWER(Command) LIKE '%/maxsize=%')
