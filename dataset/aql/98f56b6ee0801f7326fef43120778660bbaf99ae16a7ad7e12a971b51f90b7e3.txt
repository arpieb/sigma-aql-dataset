SELECT * FROM events WHERE devicetype=12 AND (LOWER("Parent Process Path") LIKE '%\powershell.exe' OR LOWER("Parent Process Path") LIKE '%\pwsh.exe') AND (LOWER("Process Path") LIKE '%\powershell.exe' OR LOWER("Process Name") LIKE '%\powershell.exe') AND LOWER(Command) LIKE '% -nologo -windowstyle minimized -file %' AND (LOWER(Command) LIKE '%\appdata\local\temp\%' OR LOWER(Command) LIKE '%\windows\temp\%')
