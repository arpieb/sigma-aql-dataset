SELECT * FROM events WHERE devicetype=12 AND (LOWER("Parent Process Path") LIKE '%\cmd.exe' OR LOWER("Parent Process Path") LIKE '%\cscript.exe' OR LOWER("Parent Process Path") LIKE '%\curl.exe' OR LOWER("Parent Process Path") LIKE '%\mshta.exe' OR LOWER("Parent Process Path") LIKE '%\powershell.exe' OR LOWER("Parent Process Path") LIKE '%\pwsh.exe' OR LOWER("Parent Process Path") LIKE '%\wscript.exe') AND (LOWER("Process Path") LIKE '%\rundll32.exe' OR LOWER("Process Name") LIKE '%\rundll32.exe') AND (LOWER(Command) LIKE '%:\programdata\%' OR LOWER(Command) LIKE '%:\users\public\%' OR LOWER(Command) LIKE '%\appdata\local\temp\%' OR LOWER(Command) LIKE '%\appdata\roaming\%') AND LOWER(Command) LIKE '%.dll%'
