SELECT * FROM events WHERE devicetype=12 AND (LOWER("Process Path") LIKE '%\powershell.exe' OR LOWER("Process Path") LIKE '%\pwsh.exe' OR LOWER("Process Name") LIKE '%\powershell.exe' OR LOWER("Process Name") LIKE '%\pwsh.exe') AND LOWER(Command) LIKE '%pester%' AND LOWER(Command) LIKE '%get-help%' OR (LOWER("Process Path") LIKE '%\cmd.exe' OR LOWER("Process Name") LIKE '%\cmd.exe') AND LOWER(Command) LIKE '%pester%' AND LOWER(Command) LIKE '%;%' AND (LOWER(Command) LIKE '%help%' OR LOWER(Command) LIKE '%?%')
