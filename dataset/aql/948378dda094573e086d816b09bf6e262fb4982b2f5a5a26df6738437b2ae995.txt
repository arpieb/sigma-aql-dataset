SELECT * FROM events WHERE devicetype=12 AND (LOWER("Process Path") LIKE '%\schtasks.exe' OR LOWER("Process Name") LIKE '%\schtasks.exe' OR Filename='schtasks.exe') AND LOWER(Command) LIKE '% /create %' AND (LOWER(Command) LIKE '%powershell%' OR LOWER(Command) LIKE '%pwsh%' OR LOWER(Command) LIKE '%cmd /c %' OR LOWER(Command) LIKE '%cmd /k %' OR LOWER(Command) LIKE '%cmd /r %' OR LOWER(Command) LIKE '%cmd.exe /c %' OR LOWER(Command) LIKE '%cmd.exe /k %' OR LOWER(Command) LIKE '%cmd.exe /r %') AND (LOWER(Command) LIKE '%c:\programdata\%' OR LOWER(Command) LIKE '%%programdata%%')
