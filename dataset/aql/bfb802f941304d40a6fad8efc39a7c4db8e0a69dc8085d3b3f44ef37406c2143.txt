SELECT * FROM events WHERE devicetype=12 AND LOWER("Parent Process Path") LIKE '%\winrar.exe' AND LOWER(Command) LIKE '%\appdata\local\temp\rar$%' AND Command MATCHES '\.[a-zA-Z0-9]{1,4} \.' AND (LOWER("Process Path") LIKE '%\cmd.exe' OR LOWER("Process Path") LIKE '%\wscript.exe' OR LOWER("Process Name") LIKE '%\cmd.exe' OR LOWER("Process Name") LIKE '%\wscript.exe' OR Filename IN('Cmd.Exe', 'cscript.exe', 'PowerShell.EXE', 'pwsh.dll', 'wscript.exe'))
