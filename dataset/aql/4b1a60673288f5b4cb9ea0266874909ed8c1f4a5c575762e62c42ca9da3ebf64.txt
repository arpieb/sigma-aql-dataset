SELECT * FROM events WHERE devicetype=12 AND (LOWER("Process Path") LIKE '%\taskkill.exe' OR LOWER("Process Name") LIKE '%\taskkill.exe' OR Filename='taskkill.exe') AND LOWER(Command) LIKE '% /f%' AND LOWER(Command) LIKE '% /im %' AND NOT(((LOWER("Parent Process Path") LIKE '%\appdata\local\temp\%' OR LOWER("Parent Process Path") LIKE '%:\windows\temp%') AND LOWER("Parent Process Path") LIKE '%.tmp'))
