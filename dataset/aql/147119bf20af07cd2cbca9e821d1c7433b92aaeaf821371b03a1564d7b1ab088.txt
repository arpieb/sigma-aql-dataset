SELECT *, Command, "Parent Command" FROM events WHERE devicetype=12 AND (LOWER("Process Path") LIKE '%\wscript.exe' OR LOWER("Process Path") LIKE '%\cscript.exe' OR LOWER("Process Name") LIKE '%\wscript.exe' OR LOWER("Process Name") LIKE '%\cscript.exe') AND (LOWER(Command) LIKE '%c:\users\%' OR LOWER(Command) LIKE '%c:\programdata\%') AND (LOWER(Command) LIKE '%.jse%' OR LOWER(Command) LIKE '%.vbe%' OR LOWER(Command) LIKE '%.js%' OR LOWER(Command) LIKE '%.vba%' OR LOWER(Command) LIKE '%.vbs%') AND NOT(LOWER("Parent Process Path") LIKE '%\winzip%')
