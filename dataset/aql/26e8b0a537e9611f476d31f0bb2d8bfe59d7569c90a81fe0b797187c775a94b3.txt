SELECT * FROM events WHERE devicetype=12 AND (LOWER("Process Path") LIKE '%\pwsh.exe' OR LOWER("Process Name") LIKE '%\pwsh.exe' OR Filename='pwsh.dll') AND (LOWER(Command) LIKE '%-securitydescriptorsddl %' OR LOWER(Command) LIKE '%-sd %') AND LOWER(Command) LIKE '%set-service %' AND LOWER(Command) LIKE '%d;;%' AND (LOWER(Command) LIKE '%;;;iu%' OR LOWER(Command) LIKE '%;;;su%' OR LOWER(Command) LIKE '%;;;ba%' OR LOWER(Command) LIKE '%;;;sy%' OR LOWER(Command) LIKE '%;;;wd%')
