SELECT * FROM events WHERE devicetype=11 AND (LOWER("Process Path") LIKE '%/curl' OR LOWER("Process Name") LIKE '%/curl') AND (LOWER(Command) LIKE '% --form%' OR LOWER(Command) LIKE '% --upload-file %' OR LOWER(Command) LIKE '% --data %' OR LOWER(Command) LIKE '% --data-%' OR Command MATCHES '\s-[FTd]\s') AND NOT((LOWER(Command) LIKE '%://localhost%' OR LOWER(Command) LIKE '%://127.0.0.1%'))
