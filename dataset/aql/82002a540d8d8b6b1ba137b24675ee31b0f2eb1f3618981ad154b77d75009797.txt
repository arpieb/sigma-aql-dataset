SELECT * FROM events WHERE devicetype=12 AND (LOWER("Process Path") LIKE '%\msdtc.exe' OR LOWER("Process Path") LIKE '%\gpsvc.exe' OR LOWER("Process Name") LIKE '%\msdtc.exe' OR LOWER("Process Name") LIKE '%\gpsvc.exe') AND NOT((LOWER("Process Path") LIKE 'c:\windows\system32\%' OR LOWER("Process Path") LIKE 'c:\windows\syswow64\%' OR LOWER("Process Name") LIKE 'c:\windows\system32\%' OR LOWER("Process Name") LIKE 'c:\windows\syswow64\%'))
