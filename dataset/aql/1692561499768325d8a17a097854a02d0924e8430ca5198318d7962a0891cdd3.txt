SELECT * FROM events WHERE devicetype=12 AND (LOWER("Process Path") LIKE '%\rundll32.exe' OR LOWER("Process Name") LIKE '%\rundll32.exe' OR Filename='RUNDLL32.EXE') AND (LOWER(Command) LIKE '%,rundll' OR LOWER(Command) LIKE '%,control_rundll') AND NOT((LOWER(Command) LIKE '%.dll,control_rundll' OR LOWER(Command) LIKE '%.dll",control_rundll' OR LOWER(Command) LIKE '%.dll'',control_rundll' OR LOWER("Parent Process Path") LIKE '%\tracker.exe'))
