SELECT * FROM events WHERE devicetype=12 AND (LOWER("Process Path") LIKE '%\net.exe' OR LOWER("Process Path") LIKE '%\net1.exe' OR LOWER("Process Name") LIKE '%\net.exe' OR LOWER("Process Name") LIKE '%\net1.exe') AND LOWER(Command) LIKE '%time%' OR (LOWER("Process Path") LIKE '%\w32tm.exe' OR LOWER("Process Name") LIKE '%\w32tm.exe') AND LOWER(Command) LIKE '%tz%'
