SELECT * FROM events WHERE devicetype=12 AND (LOWER("Process Path") LIKE '%\regsvr32.exe' OR LOWER("Process Name") LIKE '%\regsvr32.exe' OR Filename='REGSVR32.EXE') AND NOT((LOWER(Command) LIKE '%.ax%' OR LOWER(Command) LIKE '%.cpl%' OR LOWER(Command) LIKE '%.dll%' OR LOWER(Command) LIKE '%.ocx%' OR Command IS NULL OR Command='')) AND NOT((LOWER(Command) LIKE '%.ppl%' OR LOWER(Command) LIKE '%.bav%'))
