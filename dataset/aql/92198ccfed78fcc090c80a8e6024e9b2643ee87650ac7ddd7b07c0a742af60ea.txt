SELECT * FROM events WHERE devicetype=12 AND (LOWER(Command) LIKE '%\\%' AND LOWER(Command) LIKE '%$%' OR LOWER(Command) LIKE '%\sysvol\%') AND (LOWER("Process Path") LIKE '%\robocopy.exe' OR LOWER("Process Path") LIKE '%\xcopy.exe' OR LOWER("Process Name") LIKE '%\robocopy.exe' OR LOWER("Process Name") LIKE '%\xcopy.exe' OR Filename IN('robocopy.exe', 'XCOPY.EXE') OR (LOWER("Process Path") LIKE '%\cmd.exe' OR LOWER("Process Name") LIKE '%\cmd.exe' OR Filename='Cmd.Exe') AND LOWER(Command) LIKE '%copy%' OR (LOWER("Process Path") LIKE '%\powershell.exe%' OR LOWER("Process Path") LIKE '%\pwsh.exe%' OR LOWER("Process Name") LIKE '%\powershell.exe%' OR LOWER("Process Name") LIKE '%\pwsh.exe%' OR Filename IN('PowerShell.EXE', 'pwsh.dll')) AND (LOWER(Command) LIKE '%copy-item%' OR LOWER(Command) LIKE '%copy %' OR LOWER(Command) LIKE '%cpi %' OR LOWER(Command) LIKE '% cp %' OR LOWER(Command) LIKE '%move %' OR LOWER(Command) LIKE '%move-item%' OR LOWER(Command) LIKE '% mi %' OR LOWER(Command) LIKE '% mv %'))
