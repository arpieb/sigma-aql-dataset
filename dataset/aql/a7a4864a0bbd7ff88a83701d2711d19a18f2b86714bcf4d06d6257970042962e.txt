SELECT * FROM events WHERE devicetype=12 AND (LOWER("Process Path") LIKE '%\appcmd.exe' OR LOWER("Process Name") LIKE '%\appcmd.exe' OR Filename='appcmd.exe') AND LOWER(Command) LIKE '%list %' AND (LOWER(Command) LIKE '% /config%' OR LOWER(Command) LIKE '% /xml%' OR LOWER(Command) LIKE '% -config%' OR LOWER(Command) LIKE '% -xml%' OR (LOWER(Command) LIKE '% /@t%' OR LOWER(Command) LIKE '% /text%' OR LOWER(Command) LIKE '% /show%' OR LOWER(Command) LIKE '% -@t%' OR LOWER(Command) LIKE '% -text%' OR LOWER(Command) LIKE '% -show%') AND (LOWER(Command) LIKE '%:*%' OR LOWER(Command) LIKE '%password%'))
