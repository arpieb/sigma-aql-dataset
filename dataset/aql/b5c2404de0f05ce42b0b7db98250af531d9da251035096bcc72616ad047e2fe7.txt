SELECT * FROM events WHERE devicetype=12 AND (LOWER("Process Path") LIKE '%\sc.exe' OR LOWER("Process Name") LIKE '%\sc.exe') AND (LOWER(Command) LIKE '%create%' OR LOWER(Command) LIKE '%config%') AND LOWER(Command) LIKE '%binpath%' AND LOWER(Command) LIKE '%type%' AND LOWER(Command) LIKE '%kernel%'
