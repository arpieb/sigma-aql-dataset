SELECT *, Command, "Parent Command" FROM events WHERE devicetype=12 AND (LOWER("Process Path") LIKE '%\nsudo.exe' OR LOWER("Process Path") LIKE '%\nsudolc.exe' OR LOWER("Process Path") LIKE '%\nsudolg.exe' OR LOWER("Process Name") LIKE '%\nsudo.exe' OR LOWER("Process Name") LIKE '%\nsudolc.exe' OR LOWER("Process Name") LIKE '%\nsudolg.exe' OR Filename IN('NSudo.exe', 'NSudoLC.exe', 'NSudoLG.exe')) AND (LOWER(Command) LIKE '%-u:s %' OR LOWER(Command) LIKE '%-u:t %' OR LOWER(Command) LIKE '%-u:e %' OR LOWER(Command) LIKE '%-p:e %' OR LOWER(Command) LIKE '%-m:s %' OR LOWER(Command) LIKE '%-m:h %' OR LOWER(Command) LIKE '%-u=s %' OR LOWER(Command) LIKE '%-u=t %' OR LOWER(Command) LIKE '%-u=e %' OR LOWER(Command) LIKE '%-p=e %' OR LOWER(Command) LIKE '%-m=s %' OR LOWER(Command) LIKE '%-m=h %' OR LOWER(Command) LIKE '%-showwindowmode:hide%')
