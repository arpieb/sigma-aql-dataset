SELECT * FROM events WHERE devicetype=12 AND Initiated='true' AND (LOWER("Process Path") LIKE '%\regsvr32.exe' OR LOWER("Process Path") LIKE '%\rundll32.exe' OR LOWER("Process Path") LIKE '%\dllhost.exe' OR LOWER("Process Name") LIKE '%\regsvr32.exe' OR LOWER("Process Name") LIKE '%\rundll32.exe' OR LOWER("Process Name") LIKE '%\dllhost.exe') AND (LOWER(Command) LIKE '%\regsvr32.exe' OR LOWER(Command) LIKE '%\rundll32.exe' OR LOWER(Command) LIKE '%\dllhost.exe') AND NOT((Command='' OR Command IS NULL))
