SELECT *, "Parent Process Path", Command FROM events WHERE devicetype=12 AND (LOWER("Process Path") LIKE '%\regedit.exe' OR LOWER("Process Name") LIKE '%\regedit.exe' OR Filename='REGEDIT.EXE') AND (LOWER(Command) LIKE '% /e %' OR LOWER(Command) LIKE '% -e %') AND NOT(((LOWER(Command) LIKE '%hklm%' OR LOWER(Command) LIKE '%hkey_local_machine%') AND (LOWER(Command) LIKE '%\system' OR LOWER(Command) LIKE '%\sam' OR LOWER(Command) LIKE '%\security')))
