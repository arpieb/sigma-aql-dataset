SELECT * FROM events WHERE devicetype=12 AND (LOWER("Process Path") LIKE '%\cmd.exe' OR LOWER("Process Name") LIKE '%\cmd.exe' OR Filename='Cmd.Exe') AND LOWER("Parent Process Path") LIKE '%\7zfm.exe' AND NOT((LOWER(Command) LIKE '% /c %' OR LOWER(Command) LIKE '% /k %' OR LOWER(Command) LIKE '% /r %' OR Command IS NULL))
