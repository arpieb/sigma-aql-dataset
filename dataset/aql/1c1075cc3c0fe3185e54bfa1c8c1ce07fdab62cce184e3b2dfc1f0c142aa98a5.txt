SELECT * FROM events WHERE devicetype=12 AND (LOWER("Process Path") LIKE '%\cmd.exe' OR LOWER("Process Path") LIKE '%\powershell.exe' OR LOWER("Process Path") LIKE '%\pwsh.exe' OR LOWER("Process Name") LIKE '%\cmd.exe' OR LOWER("Process Name") LIKE '%\powershell.exe' OR LOWER("Process Name") LIKE '%\pwsh.exe' OR Filename IN('Cmd.Exe', 'PowerShell.EXE', 'pwsh.dll')) AND LOWER("Parent Process Path") LIKE '%\elevation_service.exe' AND "Integrity Level"='System'
