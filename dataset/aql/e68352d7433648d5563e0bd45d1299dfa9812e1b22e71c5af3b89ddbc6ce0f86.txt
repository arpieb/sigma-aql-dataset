SELECT *, Command, "Parent Command" FROM events WHERE devicetype=12 AND (LOWER("Process Path") LIKE '%\certreq.exe' OR LOWER("Process Name") LIKE '%\certreq.exe' OR Filename='CertReq.exe') AND LOWER(Command) LIKE '% -post %' AND LOWER(Command) LIKE '% -config %' AND LOWER(Command) LIKE '% http%' AND LOWER(Command) LIKE '% c:\windows\win.ini %'
