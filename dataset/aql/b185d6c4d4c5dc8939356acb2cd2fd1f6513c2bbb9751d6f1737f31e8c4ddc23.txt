SELECT * FROM events WHERE devicetype=12 AND (LOWER("Process Path") LIKE '%\find.exe' OR LOWER("Process Path") LIKE '%\findstr.exe' OR LOWER("Process Name") LIKE '%\find.exe' OR LOWER("Process Name") LIKE '%\findstr.exe' OR Filename IN('FIND.EXE', 'FINDSTR.EXE')) AND LOWER(Command) LIKE '%lsass%' OR LOWER(Command) LIKE '% /i "lsass%' OR LOWER(Command) LIKE '% /i lsass.exe%' OR LOWER(Command) LIKE '%findstr "lsass%' OR LOWER(Command) LIKE '%findstr lsass%' OR LOWER(Command) LIKE '%findstr.exe "lsass%' OR LOWER(Command) LIKE '%findstr.exe lsass%'
