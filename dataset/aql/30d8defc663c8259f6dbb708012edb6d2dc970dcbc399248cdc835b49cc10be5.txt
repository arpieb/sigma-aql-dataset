SELECT * FROM events WHERE devicetype=12 AND (LOWER("Process Path") LIKE '%\powershell.exe' OR LOWER("Process Path") LIKE '%\pwsh.exe' OR LOWER("Process Name") LIKE '%\powershell.exe' OR LOWER("Process Name") LIKE '%\pwsh.exe' OR Filename IN('PowerShell.EXE', 'pwsh.dll')) AND LOWER(Command) LIKE '%system.security.cryptography.%' AND (LOWER(Command) LIKE '%.aescryptoserviceprovider%' OR LOWER(Command) LIKE '%.descryptoserviceprovider%' OR LOWER(Command) LIKE '%.dsacryptoserviceprovider%' OR LOWER(Command) LIKE '%.rc2cryptoserviceprovider%' OR LOWER(Command) LIKE '%.rijndael%' OR LOWER(Command) LIKE '%.rsacryptoserviceprovider%' OR LOWER(Command) LIKE '%.tripledescryptoserviceprovider%')
