SELECT * FROM events WHERE devicetype=12 AND (LOWER("Process Path") LIKE '%\netsh.exe' OR LOWER("Process Name") LIKE '%\netsh.exe' OR Filename='netsh.exe') AND LOWER(Command) LIKE '%firewall %' AND LOWER(Command) LIKE '%add %' AND LOWER(Command) LIKE '%tcp %' AND Command LIKE '%3389%' AND (LOWER(Command) LIKE '%portopening%' OR LOWER(Command) LIKE '%advfirewall%' AND LOWER(Command) LIKE '%rule%' AND LOWER(Command) LIKE '%allow%')
