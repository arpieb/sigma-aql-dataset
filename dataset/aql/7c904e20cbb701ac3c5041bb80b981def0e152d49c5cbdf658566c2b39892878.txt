SELECT *, "Process Path", "Process Name", Filename FROM events WHERE devicetype=12 AND (LOWER("Process Path") LIKE '%\wscript.exe' OR LOWER("Process Path") LIKE '%\cscript.exe' OR LOWER("Process Name") LIKE '%\wscript.exe' OR LOWER("Process Name") LIKE '%\cscript.exe') AND (LOWER(Filename) LIKE 'c:\users\%' OR LOWER(Filename) LIKE 'c:\programdata%') AND (LOWER(Filename) LIKE '%.jse' OR LOWER(Filename) LIKE '%.vbe' OR LOWER(Filename) LIKE '%.js' OR LOWER(Filename) LIKE '%.vba' OR LOWER(Filename) LIKE '%.vbs')
