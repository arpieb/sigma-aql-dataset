SELECT * FROM events WHERE devicetype=12 AND (LOWER("Process Path") LIKE '%\rundll32.exe' OR LOWER("Process Name") LIKE '%\rundll32.exe' OR Filename='RUNDLL32.EXE') AND LOWER(Command) LIKE '%dllregisterserver%' AND NOT((LOWER(Command) LIKE '%:\program files (x86)%' OR LOWER(Command) LIKE '%:\program files\%' OR LOWER(Command) LIKE '%:\windows\system32\%' OR LOWER(Command) LIKE '%:\windows\syswow64\%'))
