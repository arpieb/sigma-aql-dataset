SELECT *, "Process Path", "Process Name", "Process Name", "Target Process Name", "Object Name" FROM events WHERE devicetype=12 AND ("Event ID" IN('12', '13')) AND (LOWER("Process Name") LIKE '%system\%' OR LOWER("Target Process Name") LIKE '%system\%' OR LOWER("Object Name") LIKE '%system\%') AND (LOWER("Process Name") LIKE '%\services\localnetservice' OR LOWER("Target Process Name") LIKE '%\services\localnetservice' OR LOWER("Object Name") LIKE '%\services\localnetservice')
