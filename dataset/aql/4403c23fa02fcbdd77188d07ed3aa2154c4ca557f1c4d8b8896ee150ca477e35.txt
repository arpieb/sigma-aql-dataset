SELECT * FROM events WHERE devicetype=12 AND (LOWER("Process Path") LIKE '%\sc.exe' OR LOWER("Process Name") LIKE '%\sc.exe') AND "Integrity Level"='Medium' AND (LOWER(Command) LIKE '%config%' AND LOWER(Command) LIKE '%binpath%' OR LOWER(Command) LIKE '%failure%' AND LOWER(Command) LIKE '%command%')
