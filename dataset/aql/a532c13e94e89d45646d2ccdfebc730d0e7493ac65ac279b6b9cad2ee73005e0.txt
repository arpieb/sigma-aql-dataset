SELECT * FROM events WHERE devicetype=12 AND (LOWER("Process Path") LIKE '%\regsvr32.exe' OR LOWER("Process Name") LIKE '%\regsvr32.exe' OR Filename='REGSVR32.EXE') AND (LOWER(Command) LIKE '%:\programdata\%' OR LOWER(Command) LIKE '%:\temp\%' OR LOWER(Command) LIKE '%:\users\public\%' OR LOWER(Command) LIKE '%:\windows\temp\%' OR LOWER(Command) LIKE '%\appdata\local\temp\%' OR LOWER(Command) LIKE '%\appdata\roaming\%')
