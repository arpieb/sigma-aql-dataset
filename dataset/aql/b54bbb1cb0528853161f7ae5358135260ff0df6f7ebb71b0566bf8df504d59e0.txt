SELECT * FROM events WHERE devicetype=12 AND (LOWER("Process Path") LIKE '%\tree.com' OR LOWER("Process Path") LIKE '%\wmic.exe' OR LOWER("Process Path") LIKE '%\doskey.exe' OR LOWER("Process Path") LIKE '%\sc.exe' OR LOWER("Process Name") LIKE '%\tree.com' OR LOWER("Process Name") LIKE '%\wmic.exe' OR LOWER("Process Name") LIKE '%\doskey.exe' OR LOWER("Process Name") LIKE '%\sc.exe' OR Filename IN('wmic.exe', 'DOSKEY.EXE', 'sc.exe')) AND (LOWER("Parent Command") LIKE '% > %temp%\%' OR LOWER("Parent Command") LIKE '% > %tmp%\%')
