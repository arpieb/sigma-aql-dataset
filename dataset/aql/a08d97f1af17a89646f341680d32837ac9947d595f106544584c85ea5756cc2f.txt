SELECT *, "Process Path", "Process Name", Command, "Parent Command" FROM events WHERE devicetype=12 AND LOWER("Parent Process Path") LIKE '%\taskmgr.exe' AND NOT((LOWER("Process Path") LIKE '%\resmon.exe' OR LOWER("Process Path") LIKE '%\mmc.exe' OR LOWER("Process Path") LIKE '%\taskmgr.exe' OR LOWER("Process Name") LIKE '%\resmon.exe' OR LOWER("Process Name") LIKE '%\mmc.exe' OR LOWER("Process Name") LIKE '%\taskmgr.exe'))
