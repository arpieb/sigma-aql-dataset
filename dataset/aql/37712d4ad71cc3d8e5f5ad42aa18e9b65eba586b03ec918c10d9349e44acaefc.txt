SELECT * FROM events WHERE devicetype=12 AND (LOWER(Command) LIKE '%.key%' OR LOWER(Command) LIKE '%.pgp%' OR LOWER(Command) LIKE '%.gpg%' OR LOWER(Command) LIKE '%.ppk%' OR LOWER(Command) LIKE '%.p12%' OR LOWER(Command) LIKE '%.pem%' OR LOWER(Command) LIKE '%.pfx%' OR LOWER(Command) LIKE '%.cer%' OR LOWER(Command) LIKE '%.p7b%' OR LOWER(Command) LIKE '%.asc%') AND ((LOWER("Process Path") LIKE '%\cmd.exe' OR LOWER("Process Name") LIKE '%\cmd.exe' OR Filename='Cmd.Exe') AND LOWER(Command) LIKE '%dir %' OR (LOWER("Process Path") LIKE '%\powershell.exe' OR LOWER("Process Path") LIKE '%\pwsh.exe' OR LOWER("Process Name") LIKE '%\powershell.exe' OR LOWER("Process Name") LIKE '%\pwsh.exe' OR Filename IN('PowerShell.EXE', 'pwsh.dll')) AND LOWER(Command) LIKE '%get-childitem %' OR LOWER("Process Path") LIKE '%\findstr.exe' OR LOWER("Process Name") LIKE '%\findstr.exe' OR Filename='FINDSTR.EXE')
