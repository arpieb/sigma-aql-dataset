SELECT *, Username, Command, "Parent Command" FROM events WHERE devicetype=12 AND (LOWER("Process Path") LIKE '%\net.exe' OR LOWER("Process Path") LIKE '%\net1.exe' OR LOWER("Process Name") LIKE '%\net.exe' OR LOWER("Process Name") LIKE '%\net1.exe' OR Filename IN('net.exe', 'net1.exe')) AND (LOWER(Command) LIKE '% group%' OR LOWER(Command) LIKE '% localgroup%' OR LOWER(Command) LIKE '% user%' OR LOWER(Command) LIKE '% view%' OR LOWER(Command) LIKE '% share%' OR LOWER(Command) LIKE '% accounts%' OR LOWER(Command) LIKE '% stop %' OR LOWER(Command) LIKE '% start%')
