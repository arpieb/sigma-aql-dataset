SELECT * FROM events WHERE devicetype=12 AND (LOWER("Process Path") LIKE '%\netsh.exe' OR LOWER("Process Name") LIKE '%\netsh.exe' OR Filename='netsh.exe') AND (LOWER(Command) LIKE '%firewall%' AND LOWER(Command) LIKE '%set%' AND LOWER(Command) LIKE '%opmode%' AND LOWER(Command) LIKE '%disable%' OR LOWER(Command) LIKE '%advfirewall%' AND LOWER(Command) LIKE '%set%' AND LOWER(Command) LIKE '%state%' AND LOWER(Command) LIKE '%off%')
