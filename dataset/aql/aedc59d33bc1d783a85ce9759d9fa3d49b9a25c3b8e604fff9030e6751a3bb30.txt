SELECT * FROM events WHERE devicetype=12 AND (Filename IN('wscript.exe', 'cscript.exe') OR LOWER("Process Path") LIKE '%\wscript.exe' OR LOWER("Process Path") LIKE '%\cscript.exe' OR LOWER("Process Name") LIKE '%\wscript.exe' OR LOWER("Process Name") LIKE '%\cscript.exe') AND (LOWER(Command) LIKE '%.csv%' OR LOWER(Command) LIKE '%.dat%' OR LOWER(Command) LIKE '%.doc%' OR LOWER(Command) LIKE '%.gif%' OR LOWER(Command) LIKE '%.jpeg%' OR LOWER(Command) LIKE '%.jpg%' OR LOWER(Command) LIKE '%.png%' OR LOWER(Command) LIKE '%.ppt%' OR LOWER(Command) LIKE '%.txt%' OR LOWER(Command) LIKE '%.xls%' OR LOWER(Command) LIKE '%.xml%')
